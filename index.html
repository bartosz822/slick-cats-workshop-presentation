							abstract class IdRepository[Entity, Table <: IdTable[Entity]](val query: TableQuery[Table]) {

								protected def byId = Compiled { (id: Rep[Long]) => query.filter(_.id === id) }

								def save(entity: Entity): DBIO[Long] = query.returning(query.map(_.id)) += entity
								def findById(id: Long): DBIO[Option[Entity]] = byId(id).result.headOption
								def findAll(): DBIO[Seq[Entity]] = query.result
								def deleteById(id: Long): DBIO[Unit] = byId(id).delete >> DBIO.successful(())
							}
